   dayfractoday = function()
   xh=os.date("!%H")
   xm=os.date("!%M")
   xs=os.date("!%S")   
   carryover=0
   if math.tointeger (xh) >= 12
   then
      carryover = 1
       xh=xh-12
		else
		   carryover = -1
      xh=xh+12
 		end
   result = xh/24 + xm/(24*60) + xs/(24*60*60)
--   s = string.format("%.4f",result)

   return result
    end

\newcommand\findjdnd[3]{%
  \directlua{
    tex.sprint(JDNd(#1, #2, #3))
